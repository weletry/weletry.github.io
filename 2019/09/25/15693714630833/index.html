# 获取所有目录的uuid及名字
def get_all_folder(p_uuid, folder_list):
    conn = sqlite3.connect(file_path+"mainlib.db")
    cursor = conn.cursor()
    sql = " select id, uuid,name from cat where pid ="+str(p_uuid)
    cursor.execute(sql)
    values = cursor.fetchall()


    for item in values:
        get_all_folder(item[1], folder_list)
        # 文档目录下的文章
        a_sql =  "select ca.rid,a.uuid,a.dateAdd, a.dateModif from cat_article ca join article a on ca.aid = a.uuid where ca.rid=" + str(item[1])
        cursor.execute(a_sql)
        cat_article_lists = cursor.fetchall()
        articles = []
        for article in cat_article_lists:

            # 查询tag
            tags = []
            tag_sql = "select t.name from tag_article ta join tag t on ta.rid = t.id where ta.aid = "+str (article[1])
            cursor.execute(tag_sql)
            tag_article_lists = cursor.fetchall()
            for tag in tag_article_lists:
                tags.append(tag[0])
            articles.append({
                "uuid": article[1],
                "dateAdd": article[2],
                "dateModif": article[3],
                "tags": tags
            })

        folder_list.append({
            "id":item[0],
            "uuid": item[1],
            "name": item[2],
            "articles":articles
        })

    # 保存所有文件夹，然后再向下递归
    return folder_list
